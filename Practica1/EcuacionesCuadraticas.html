package com.mycompany.ejercicio2; import java.util.Scanner; /** * * @author usuario */ public class Ejercicio2 { public static void main(String[] args) { Scanner sc = new Scanner(System.in); System.out.println("Ejercicio 2"); System.out.print("Ingrese a, b, c: "); double a = sc.nextDouble(); double b = sc.nextDouble(); double c = sc.nextDouble(); EcuancionCuadratica eq = new EcuancionCuadratica(a, b, c); double disc = eq.getDiscriminante(); if (disc > 0) { System.out.println("La ecuacion tiene dos raices: " + eq.getRaiz1() + " y " + eq.getRaiz2()); } else if (disc == 0) { System.out.println("La ecuacion tiene una raiz: " + eq.getRaiz1()); } else { System.out.println("La ecuacion no tiene raices reales"); } } } package com.mycompany.ejercicio2; import java.util.Scanner; public class EcuancionCuadratica { private double a, b, c; public EcuancionCuadratica(double a, double b, double c) { this.a = a; this.b = b; this.c = c; } public double getDiscriminante() { return b * b - 4 * a * c; } public double getRaiz1() { if (getDiscriminante() >= 0) { return (-b + Math.sqrt(getDiscriminante())) / (2 * a); } return 0; } public double getRaiz2() { if (getDiscriminante() >= 0) { return (-b - Math.sqrt(getDiscriminante())) / (2 * a); } return 0; } }
